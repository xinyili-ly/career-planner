## 一、接口总览

| 序号 | 前端函数 | HTTP方法 | 后端路径 | 用途 | 必填参数 |
|:---:|---------|:-------:|---------|------|---------|
| 1 | `fetchRecommendJobList()` | GET | `/recommend/list` | 获取岗位列表 | 无 |
| 2 | `postMatchDetail()` | POST | `/api/v1/match/detail` | 获取匹配详情 | `job_id` |
| 3 | `postMatchReport()` | POST | `/api/v1/match/report` | 生成完整职业报告（含module_4） | `job_id` |
| 4 | `postMatchReportPreview()` | POST | `/api/v1/match/report-preview` | 报告预览（仅module_1~3） | `job_id` |
| 5 | `postMatchReportFinalize()` | POST | `/api/v1/match/report/finalize` | 最终报告润色（仅module_4） | `job_id` + `preview_data` |
| 6 | `getCurrentActionPlan()` | GET | `/api/v1/action-plan/current` | 获取当月行动计划 | `student_id` |
| 7 | `updateActionItemStatus()` | POST | `/api/v1/action-plan/items/{item_id}/update` | 更新任务状态 | `item_id` + `body` |
| 8 | `getLatestSnapshot()` | GET | `/api/v1/profile/snapshots/latest` | 获取最新画像快照 | `student_id` |
| 9 | `getProfileFluctuation()` | GET | `/api/v1/profile/fluctuation` | 获取画像波动详情 | `student_id` |
| 10 | `triggerMonthlyEvaluation()` | POST | `/api/v1/action-plan/evaluate` | 触发月度评估 | `body` |
| 11 | `getQuarterlyReport()` | GET | `/api/v1/reports/quarterly` | 获取季度成长报告 | `student_id` |

## 二、API 封装示例代码

### 2.1 careerAgentApi.js（推荐使用）

```javascript
// frontend/src/api/careerAgentApi.js

/**
 * 职业规划智能体后端（OpenAPI）
 * 默认：http://192.168.31.93:8002
 */
const DEFAULT_BASE = 'http://192.168.31.93:8002'

export function getCareerAgentBaseUrl() {
  const u = import.meta.env.VITE_CAREER_AGENT_API_BASE
  if (u && String(u).trim()) return String(u).replace(/\/$/, '')
  return DEFAULT_BASE
}

function baseUrlOrDefault() {
  return getCareerAgentBaseUrl() || DEFAULT_BASE
}

function parseResponse(res) {
  if (!res.ok) {
    return res.json().then(err => Promise.reject(new Error(err?.detail || `HTTP ${res.status}`)))
  }
  return res.json()
}

/**
 * 1. 获取岗位列表
 * @returns {Promise<Array>} 岗位列表
 */
export async function fetchRecommendJobList() {
  const url = new URL('/recommend/list', baseUrlOrDefault())
  const res = await fetch(url.toString(), {
    method: 'GET',
    headers: { Accept: 'application/json' },
  })
  const data = await parseResponse(res)
  return data?.data || []
}

/**
 * 2. 获取匹配详情
 * @param {Object} body - 请求体
 * @param {string} body.job_id - 岗位ID（必填）
 * @param {string} [body.profile_cache_id] - 画像缓存ID（推荐）
 * @param {Object} [body.student_profile] - 学生画像对象（二选一）
 * @returns {Promise<Object>} 匹配详情
 */
export async function postMatchDetail(body) {
  const url = new URL('/api/v1/match/detail', baseUrlOrDefault())
  const res = await fetch(url.toString(), {
    method: 'POST',
    headers: {
      Accept: 'application/json',
      'Content-Type': 'application/json'
    },
    body: JSON.stringify(body),
  })
  return parseResponse(res)
}

/**
 * 3. 生成完整职业报告（一次性生成 module_1~4）
 * @param {Object} body - 请求体
 * @param {string} body.job_id - 岗位ID（必填）
 * @param {string} [body.profile_cache_id] - 画像缓存ID（推荐）
 * @param {Object} [body.student_profile] - 学生画像对象（二选一）
 * @param {Object} [body.plan_preferences] - 计划偏好设置
 * @param {Object} [body.precomputed_module_1] - 预计算的 module_1（可选，跳过 module_1 LLM 调用）
 * @returns {Promise<Object>} 完整报告数据
 */
export async function postMatchReport(body) {
  const url = new URL('/api/v1/match/report', baseUrlOrDefault())
  const res = await fetch(url.toString(), {
    method: 'POST',
    headers: {
      Accept: 'application/json',
      'Content-Type': 'application/json'
    },
    body: JSON.stringify(body),
  })
  return parseResponse(res)
}

/**
 * 4. 报告预览（仅 module_1~3，不含最终润色报告）
 * @param {Object} body - 请求体
 * @param {string} body.job_id - 岗位ID（必填）
 * @param {string} [body.profile_cache_id] - 画像缓存ID（推荐）
 * @param {Object} [body.student_profile] - 学生画像对象（二选一）
 * @param {Object} [body.plan_preferences] - 计划偏好设置
 * @returns {Promise<Object>} 报告预览数据
 */
export async function postMatchReportPreview(body) {
  const url = new URL('/api/v1/match/report-preview', baseUrlOrDefault())
  const res = await fetch(url.toString(), {
    method: 'POST',
    headers: {
      Accept: 'application/json',
      'Content-Type': 'application/json'
    },
    body: JSON.stringify(body),
  })
  return parseResponse(res)
}

/**
 * 5. 最终报告润色（仅生成 module_4）
 * @param {Object} body - 请求体
 * @param {string} body.job_id - 岗位ID（必填）
 * @param {string} [body.profile_cache_id] - 画像缓存ID（推荐）
 * @param {Object} [body.student_profile] - 学生画像对象（二选一）
 * @param {Object} body.preview_data - 上一步 report-preview 返回的预览数据（**必填**）
 * @returns {Promise<Object>} 仅包含 module_4
 */
export async function postMatchReportFinalize(body) {
  const url = new URL('/api/v1/match/report/finalize', baseUrlOrDefault())
  const res = await fetch(url.toString(), {
    method: 'POST',
    headers: {
      Accept: 'application/json',
      'Content-Type': 'application/json'
    },
    body: JSON.stringify(body),
  })
  return parseResponse(res)
}

/**
 * 6. 获取当月行动计划
 * @param {string} studentId - 学生ID（profile_cache_id）
 * @returns {Promise<Object>} 当月行动计划（含 pending/completed/rolled_over 分组）
 */
export async function getCurrentActionPlan(studentId) {
  const url = new URL('/api/v1/action-plan/current', baseUrlOrDefault())
  url.searchParams.set('student_id', studentId)
  const res = await fetch(url.toString(), {
    method: 'GET',
    headers: { Accept: 'application/json' },
  })
  return parseResponse(res)
}

/**
 * 7. 更新任务状态
 * @param {string} itemId - 任务ID
 * @param {Object} body - 请求体
 * @param {string} body.student_id - 学生ID
 * @param {string} body.status - 新状态：pending / in_progress / completed
 * @param {string} [body.evidence] - 完成证据（可选，500字以内）
 * @returns {Promise<Object>} 更新结果（含维度变化量）
 */
export async function updateActionItemStatus(itemId, body) {
  const url = new URL(`/api/v1/action-plan/items/${itemId}/update`, baseUrlOrDefault())
  const res = await fetch(url.toString(), {
    method: 'POST',
    headers: {
      Accept: 'application/json',
      'Content-Type': 'application/json'
    },
    body: JSON.stringify(body),
  })
  return parseResponse(res)
}

/**
 * 8. 获取最新画像快照
 * @param {string} studentId - 学生ID
 * @returns {Promise<Object>} 最新快照（含7维度分和匹配分）
 */
export async function getLatestSnapshot(studentId) {
  const url = new URL('/api/v1/profile/snapshots/latest', baseUrlOrDefault())
  url.searchParams.set('student_id', studentId)
  const res = await fetch(url.toString(), {
    method: 'GET',
    headers: { Accept: 'application/json' },
  })
  return parseResponse(res)
}

/**
 * 9. 获取画像波动详情
 * @param {string} studentId - 学生ID
 * @param {string} [month] - 指定月份 YYYY-MM，不传则取最新
 * @returns {Promise<Object>} 波动详情（含维度变化原因）
 */
export async function getProfileFluctuation(studentId, month) {
  const url = new URL('/api/v1/profile/fluctuation', baseUrlOrDefault())
  url.searchParams.set('student_id', studentId)
  if (month) url.searchParams.set('month', month)
  const res = await fetch(url.toString(), {
    method: 'GET',
    headers: { Accept: 'application/json' },
  })
  return parseResponse(res)
}

/**
 * 10. 触发月度评估
 * @param {Object} body - 请求体
 * @param {string} body.student_id - 学生ID
 * @param {string} body.target_job_id - 目标岗位ID
 * @param {boolean} [body.force_rescore] - 是否强制AI重打分
 * @returns {Promise<Object>} 评估结果
 */
export async function triggerMonthlyEvaluation(body) {
  const url = new URL('/api/v1/action-plan/evaluate', baseUrlOrDefault())
  const res = await fetch(url.toString(), {
    method: 'POST',
    headers: {
      Accept: 'application/json',
      'Content-Type': 'application/json'
    },
    body: JSON.stringify(body),
  })
  return parseResponse(res)
}

/**
 * 11. 获取季度成长报告
 * @param {string} studentId - 学生ID
 * @param {number} [months] - 统计月数，默认3个月
 * @returns {Promise<Object>} 季度报告（含维度趋势分析）
 */
export async function getQuarterlyReport(studentId, months = 3) {
  const url = new URL('/api/v1/reports/quarterly', baseUrlOrDefault())
  url.searchParams.set('student_id', studentId)
  url.searchParams.set('months', String(months))
  const res = await fetch(url.toString(), {
    method: 'GET',
    headers: { Accept: 'application/json' },
  })
  return parseResponse(res)
}

/**
 * 数据映射函数：规范岗位列表项（新版，后端返回 id 而非 job_id）
 * @param {Object} item - 后端返回的岗位对象
 * @returns {Object} 前端使用的岗位对象
 */
export function normalizeJobRow(item) {
  if (!item || !item.id) return null
  return {
    id: item.id,  // 后端直接返回 id
    title: item.title || '',
    category: (item.field_tags || []).slice(0, 3).join('/'),
    summary: item.title || '',
    company: item.company || '',
    salary_range: item.salary_range || '',
    hot_index: item.hot_index || 0,
  }
}

/**
 * 行动计划映射函数（新版）
 * @param {Object} item - 后端返回的行动计划项
 * @returns {Object} 前端使用的行动计划项
 */
export function mapPlanRow(item) {
  if (!item) return null
  return {
    id: item.id || item.item_id,
    title: item.title || item.skill || '',
    description: item.description || '',
    target_dimension: item.target_dimension || '',
    target_dimension_label: item.target_dimension_label || '',
    skill_keywords: item.skill_keywords || [],
    completion_criteria: item.completion_criteria || item.criteria || '',
    estimated_hours: item.estimated_hours || 0,
    difficulty: item.difficulty || 'standard',
    milestone_checkpoints: item.milestone_checkpoints || item.milestones || [],
    status: item.status || 'pending',
    is_rolled_over: item.is_rolled_over || false,
    completed_at: item.completed_at || null,
    progress_percent: item.progress_percent || 0,
  }
}

/**
 * 报告数据标准化函数
 * @param {Object} raw - 后端返回的原始报告数据
 * @returns {Object} 前端标准化的报告数据
 */
export function normalizeMatchReportPayload(raw) {
  if (!raw) return null
  return {
    summary: raw.summary || {},
    job_info: raw.job_info || {},
    match_summary: raw.match_summary || {},
    career_report: raw.career_report || {},
    // 新版 action_plan 结构：pending_list / completed_list / rolled_over_list
    action_plan: raw.action_plan || {},
  }
}
```

## 三、接口详细说明

### 3.1 获取岗位列表 `/recommend/list`

**功能**：获取系统中所有岗位的摘要信息列表，用于第二分界面岗位选择列表。

---

#### 请求

```
GET /recommend/list
Host: http://192.168.31.93:8002
Accept: application/json
```

**请求参数**：无

---

#### 响应

```json
{
  "code": 0,
  "message": "ok",
  "apiVersion": "1",
  "data": [
    {
      "id": "JOB_001",
      "title": "后端开发工程师",
      "field_tags": ["Python", "FastAPI", "MySQL"],
      "company": "示例公司",
      "salary_range": "10000-18000元/月",
      "hot_index": 85
    },
    {
      "id": "JOB_002",
      "title": "前端开发工程师",
      "field_tags": ["Vue", "React", "TypeScript"],
      "company": "示例公司",
      "salary_range": "10000-16000元/月",
      "hot_index": 88
    }
  ]
}
```

---

#### 字段说明

| 字段 | 类型 | 说明 |
|------|------|------|
| `code` | int | 响应码，0 表示成功 |
| `message` | string | 状态信息 |
| `data` | array | 岗位列表数组 |
| `data[].id` | string | 岗位唯一标识（后端存储为 `job_id`，前端直接使用 `id`） |
| `data[].title` | string | 岗位名称 |
| `data[].field_tags` | array | 技能标签数组（前端映射为 `category`） |
| `data[].company` | string | 示例公司（后端占位字段） |
| `data[].salary_range` | string | 薪资范围（后端占位字段） |
| `data[].hot_index` | int | 热门指数（后端占位字段） |

---

#### 前端调用示例

```javascript
import { fetchRecommendJobList, normalizeJobRow } from '@/api/careerAgentApi'

const recommendedJobs = ref([])

async function loadRecommendJobs() {
  try {
    const list = await fetchRecommendJobList()
    const mapped = list.map(normalizeJobRow).filter(Boolean)
    if (mapped.length > 0) {
      recommendedJobs.value = mapped
      selectedJobId.value = mapped[0].id  // 默认选中第一个岗位
    }
  } catch (error) {
    console.error('加载岗位列表失败:', error)
    ElMessage.error('加载岗位列表失败，请稍后重试')
  }
}
```

### 3.2 获取匹配详情 `/api/v1/match/detail`

**功能**：点击岗位「详情」按钮时调用，返回规则分计算结果 + LLM 生成的详细差距分析与提升建议。用于第二分界面详情弹窗展示。

---

#### 请求

```
POST /api/v1/match/detail
Host: http://192.168.31.93:8002
Content-Type: application/json
Accept: application/json
```

**请求体**：

```json
{
  "job_id": "JOB_001",
  "profile_cache_id": "29865f50-387c-4063-9aba-24b91d9c9498"
}
```

| 字段 | 类型 | 必填 | 说明 |
|------|------|:---:|------|
| `job_id` | string | **是** | 目标岗位 ID |
| `profile_cache_id` | string | 二选一 | 画像缓存 ID（**推荐**，由阶段一简历解析返回） |
| `student_profile` | object | 二选一 | 完整学生画像对象 |

> **注意**：`profile_cache_id` 和 `student_profile` 二选一，推荐传 `profile_cache_id`。

---

```javascript
const step = ref('detail')  // 'detail' | 'report'

async function generateReport() {
  if (!selectedJobId.value) {
    ElMessage.warning('请先选择岗位')
    return
  }

  const payload = { job_id: selectedJobId.value }
  
  if (profileCacheId.value) {
    // 优先使用 profile_cache_id
    payload.profile_cache_id = profileCacheId.value
  } else if (studentProfile.value) {
    // 兜底：使用完整 student_profile
    payload.student_profile = studentProfile.value
  }

  try {
    const raw = await postMatchReport(payload)
    const normalized = normalizeMatchReportPayload(raw)
    
    if (normalized) {
      // 应用报告数据到页面
      applyNormalizedReport(normalized)
      // 同步草稿数据
      syncDraftFromCurrent()
    }
    
    // 切换到报告页面
    step.value = 'report'
    ElMessage.success('已生成职业发展规划报告')

    console.error('生成报告失败:', error)
    ElMessage.error(`生成报告失败：${error?.message || '请稍后重试'}`)
  }
}

// 应用报告数据
function applyNormalizedReport(report) {
  // 摘要信息
  summaryData.value = report.summary
  
  // 岗位信息
  jobInfoData.value = report.job_info

  matchSummaryData.value = report.match_summary

  module1Data.value = report.career_report?.module_1
  
  // 模块二：岗位画像与路径
  module2Data.value = report.career_report?.module_2
  
  // 模块三：技能学习计划
  module3Data.value = report.career_report?.module_3
  
  // 模块四：完整报告（Markdown）
  module4Data.value = report.career_report?.module_4
  
  // 行动计划（TodoList）- 新版结构含 pending_list/completed_list/rolled_over_list
  if (report.action_plan) {
    const { pending_list = [], completed_list = [], rolled_over_list = [] } = report.action_plan
    const allItems = [...pending_list, ...completed_list, ...rolled_over_list]
    actionPlanItems.value = allItems.map(mapPlanRow)
    planCompletionRate.value = report.action_plan.completion_rate
  }
}
```

## 八、注意事项

1. **API 基础地址**：
   - 开发环境默认：`http://192.168.31.93:8002`
   - 可通过 `VITE_CAREER_AGENT_API_BASE` 环境变量覆盖

2. **profile_cache_id 传递**：
   - 阶段一简历解析后返回 `profile_cache_id`
   - 阶段二、三接口优先传递此 ID，而非完整对象
   - 缓存过期时间约 24 小时

3. **错误处理**：
   - 404 错误通常表示缓存过期，需要重新上传简历
   - 500 错误记录日志并提示用户稍后重试

4. **性能优化**：
   - 切换岗位时调用 `postMatchDetail`，使用 `profile_cache_id` 可减少数据传输
   - 生成报告 `postMatchReport` 可能耗时较长（涉及多个 LLM 调用），建议显示 loading
